# esco_ingestion.py
# loads ESCO data into postgres

import logging
import pandas as pd
from sqlalchemy import create_engine
import os
from pathlib import Path

log = logging.getLogger(__name__)

DB_URL = os.getenv("DB_URL", "postgresql:///skillmatch")
engine = create_engine(DB_URL)

BASE_DIR = Path(__file__).resolve().parent.parent
ESCO_DIR = BASE_DIR / "data" / "esco"


def normalize(df):
    df.columns = (
        df.columns
        .str.strip()
        .str.lower()
        .str.replace("-", "_")
        .str.replace(" ", "_")
    )
    return df


def parse_array(val):
    """Convert newline-separated ESCO label strings to Python list for TEXT[] columns."""
    if pd.isna(val) or str(val).strip() == "":
        return None
    return [v.strip() for v in str(val).split("\n") if v.strip()]


def load_esco_occupations():
    path = ESCO_DIR / "occupations_en.csv"
    df = pd.read_csv(path)
    df = normalize(df)

    df = df.rename(columns={
        "concepturi":     "concept_uri",
        "concepttype":    "concept_type",
        "iscogroup":      "isco_group",
        "preferredlabel": "preferred_label",
        "altlabels":      "alt_labels",
        "hiddenlabels":   "hidden_labels",
        "modifieddate":   "modified_date",
        "scopenote":      "scope_note",
        "nacecode":       "nace_code",
    })

    # _en.csv files can still have duplicate concept_uri rows (e.g. one per inScheme entry).
    # Keep only the first occurrence per concept_uri — all rows for the same URI
    # have identical content since this is the English-only export.
    df = df.drop_duplicates("concept_uri", keep="first")

    df["alt_labels"]    = df["alt_labels"].apply(parse_array)
    df["hidden_labels"] = df["hidden_labels"].apply(parse_array)

    # Guard: only select columns that are actually present in this file
    want = [
        "concept_uri", "concept_type", "isco_group", "preferred_label",
        "alt_labels", "hidden_labels", "status", "modified_date",
        "scope_note", "definition", "description", "code", "nace_code"
    ]
    df = df[[c for c in want if c in df.columns]]

    df.to_sql("esco_occupations", engine, if_exists="append", index=False, method="multi")
    log.info(f"esco_occupations loaded: {len(df)} rows")


def load_esco_skills():
    path = ESCO_DIR / "skills_en.csv"
    df = pd.read_csv(path)
    df = normalize(df)

    df = df.rename(columns={
        "concepturi":     "concept_uri",
        "concepttype":    "concept_type",
        "skilltype":      "skill_type",
        "reuselevel":     "reuse_level",
        "preferredlabel": "preferred_label",
        "altlabels":      "alt_labels",
        "hiddenlabels":   "hidden_labels",
        "modifieddate":   "modified_date",
        "scopenote":      "scope_note",
    })

    df = df.drop_duplicates("concept_uri", keep="first")

    df["alt_labels"]    = df["alt_labels"].apply(parse_array)
    df["hidden_labels"] = df["hidden_labels"].apply(parse_array)

    want = [
        "concept_uri", "concept_type", "skill_type", "reuse_level",
        "preferred_label", "alt_labels", "hidden_labels", "status",
        "modified_date", "scope_note", "definition", "description"
    ]
    df = df[[c for c in want if c in df.columns]]

    df.to_sql("esco_skills", engine, if_exists="append", index=False, method="multi")
    log.info(f"esco_skills loaded: {len(df)} rows")


def load_esco_relations():
    path = ESCO_DIR / "occupationSkillRelations_en.csv"
    df = pd.read_csv(path)
    df = normalize(df)

    df = df.rename(columns={
        "occupationuri":   "occupation_uri",
        "occupationlabel": "occupation_label",
        "relationtype":    "relation_type",
        "skilltype":       "skill_type",
        "skilluri":        "skill_uri",
        "skilllabel":      "skill_label",
    })

    want = [
        "occupation_uri", "occupation_label", "relation_type",
        "skill_type", "skill_uri", "skill_label"
    ]
    df = df[[c for c in want if c in df.columns]]

    # Schema has UNIQUE(occupation_uri, skill_uri)
    df = df.drop_duplicates(["occupation_uri", "skill_uri"])

    df.to_sql("esco_occupation_skill_relations", engine, if_exists="append", index=False, method="multi")
    log.info(f"esco_occupation_skill_relations loaded: {len(df)} rows")


def run_esco():
    log.info("Loading ESCO...")
    load_esco_occupations()
    load_esco_skills()
    load_esco_relations()
    log.info("ESCO complete")
